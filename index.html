import qrcode from PIL import Image, ImageDraw, ImageFont # 🌟 LINK WEB CHÍNH CHỦ ĐÃ ĐỔI CHỮ CỦA BẠN 🌟 qr_data = "https://fbbbb2004-pixel.github.io/" qr = qrcode.QRCode( version=None, error_correction=qrcode.constants.ERROR_CORRECT_H, box_size=10, border=4, ) qr.add_data(qr_data) qr.make(fit=True) img = qr.make_image(fill_color="#0f3460", back_color="#e8f4f8").convert('RGBA') width, height = img.size overlay = Image.new('RGBA', img.size, (255, 255, 255, 0)) draw = ImageDraw.Draw(overlay) center_x, center_y = width // 2, height // 2 heart_size = 75 circle_radius = heart_size // 2 + 5 draw.ellipse([center_x - circle_radius, center_y - circle_radius, center_x + circle_radius, center_y + circle_radius], fill=(232, 244, 248, 255)) def draw_heart(draw_obj, cx, cy, size, color): s = size // 2 draw_obj.ellipse([cx - s, cy - s//2, cx, cy + s//2], fill=color) draw_obj.ellipse([cx, cy - s//2, cx + s, cy + s//2], fill=color) draw_obj.polygon([(cx - s, cy), (cx + s, cy), (cx, cy + s)], fill=color) draw_heart(draw, center_x, center_y - 5, heart_size, (255, 107, 157, 255)) draw_heart(draw, center_x, center_y - 5, heart_size - 10, (255, 255, 255, 255)) draw_heart(draw, center_x, center_y - 5, heart_size - 18, (255, 107, 157, 255)) img = Image.alpha_composite(img, overlay) border_size = 20 border_color = (255, 182, 193, 255) bordered_img = Image.new('RGBA', (width + 2*border_size, height + 2*border_size), border_color) bordered_img.paste(img, (border_size, border_size)) final_img = Image.new('RGBA', (bordered_img.width, bordered_img.height + 60), border_color) final_img.paste(bordered_img, (0, 0)) try: font = ImageFont.truetype("arial.ttf", 22) except: font = ImageFont.load_default() draw_final = ImageDraw.Draw(final_img) text = "💕 Quét để nhận điều bất ngờ 💕" bbox = draw_final.textbbox((0, 0), text, font=font) text_width = bbox[2] - bbox[0] text_x = (final_img.width - text_width) // 2 draw_final.text((text_x, bordered_img.height + 15), text, fill=(15, 52, 96, 255), font=font) # Tên file ảnh QR lưu trong máy bạn output_filename = "qr_anh_yeu_em.png" final_img.save(output_filename) print(f"🎉 XONG RỒI! File ảnh đã lưu thành công tên là: {output_filename}")
